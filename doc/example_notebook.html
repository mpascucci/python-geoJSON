Plotting data¶

These properties can be set directly in the geoJSON_feature properties

  • "color" : edge color
  • "weight" : line weight
  • "opacity" : edge opacity
  • "fillColor" : fill color
  • "fillOpacity" : fill opacity

# Now, set plot properties as a functon of the population data

# create a custom colormap
cmap = mpl.colors.LinearSegmentedColormap.from_list('y-r',['yellow','red'],N=256)

# set the plotting properties based on the population data
for f in features:
    region_name = f.properties['NAME_1']
    population = region_data[region_data.Name == region_name].Population.iloc[0]
    f.properties['fillColor'] = geoJ.hex_color(cmap(population/population_max))

# final plot
for f in features:
  geoJ.plot_geoJSON_feature(f)  

# add a colorbar
norm = mpl.colors.Normalize(vmin=0, vmax=population_max)
plt.colorbar(mpl.cm.ScalarMappable(norm=norm, cmap=cmap), ax=plt.gca(), shrink=0.6)
plt.axis("off")
plt.title("Population",fontdict={"size":18});
